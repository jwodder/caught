CREATE TABLE pokemon (dexno INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT)
-- Both columns are unique

CREATE TABLE pokemon_names (dexno INTEGER, name TEXT)
-- names are unique and lowercase
-- dexno is a foreign key to `pokemon`

CREATE TABLE games (gameID INTEGER PRIMARY KEY AUTOINCREMENT, version TEXT, player_name TEXT, dexsize INTEGER)
-- `gameID` and `player_name` columns are both unique

CREATE TABLE game_names (gameID INTEGER, name TEXT)
-- names are unique and lowercase
-- gameID is a foreign key to `games`

CREATE TABLE caught (gameID INTEGER, dexno INTEGER, status INTEGER)
-- (gameID, dexno) is the key.
